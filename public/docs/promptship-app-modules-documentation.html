Subdomain Architecture

app.promptship.dev
🌐 Subdomain Strategy

The app uses a separate subdomain (app.promptship.dev) instead of a path prefix (promptship.dev/app). This enables better separation of concerns, independent scaling, and cleaner cookie handling for authentication.

2.3 Dashboard Data Fetching

// src/app/app/page.tsx (Dashboard)

import { getSession } from '@/lib/auth';
import { db } from '@/lib/db';
import { WelcomeBanner } from '@/components/dashboard/welcome-banner';
import { QuickActions } from '@/components/dashboard/quick-actions';
import { CourseProgressCard } from '@/components/dashboard/course-progress-card';
import { RecentGenerations } from '@/components/dashboard/recent-generations';
import { WelcomeVideo } from '@/components/dashboard/welcome-video';
import { StatsCards } from '@/components/dashboard/stats-cards';
import { NewPromptsWidget } from '@/components/dashboard/new-prompts-widget';

async function getDashboardData(userId: string) {
  const [stats, recentGenerations, courseProgress, newPrompts] = await Promise.all([
    // User stats
    db.query.users.findFirst({
      where: eq(users.id, userId),
      columns: { credits: true, tier: true },
    }),
    
    // Recent generations (last 6)
    db.query.generations.findMany({
      where: eq(generations.userId, userId),
      orderBy: [desc(generations.createdAt)],
      limit: 6,
    }),
    
    // Course progress
    db.query.lessonProgress.findMany({
      where: eq(lessonProgress.userId, userId),
      with: { lesson: { with: { module: true } } },
    }),
    
    // New prompts this week
    db.query.prompts.findMany({
      where: and(
        eq(prompts.isPublished, true),
        gte(prompts.createdAt, subDays(new Date(), 7))
      ),
      orderBy: [desc(prompts.createdAt)],
      limit: 5,
    }),
  ]);
  
  return { stats, recentGenerations, courseProgress, newPrompts };
}

export default async function DashboardPage() {
  const session = await getSession();
  const data = await getDashboardData(session.user.id);
  
  return (
    <div className="grid grid-cols-1 lg:grid-cols-[1fr_320px] gap-6">
      {/* Main Content */}
      <div className="space-y-6">
        <WelcomeBanner user={session.user} />
        <QuickActions tier={session.user.tier} />
        <CourseProgressCard progress={data.courseProgress} />
        <RecentGenerations generations={data.recentGenerations} />
      </div>
      
      {/* Sidebar */}
      <div className="space-y-5">
        <WelcomeVideo />
        <StatsCards stats={data.stats} />
        <NewPromptsWidget prompts={data.newPrompts} />
      </div>
    </div>
  );
}

3.3 Prompt Card Component

// src/components/prompts/prompt-card.tsx

'use client';

import { useState } from 'react';
import Image from 'next/image';
import { Heart, Copy, Sparkles, Check } from 'lucide-react';
import { cn } from '@/lib/utils/cn';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { useToast } from '@/components/ui/use-toast';
import { useCopyPrompt } from '@/hooks/use-copy-prompt';
import { useFavorite } from '@/hooks/use-favorite';

interface PromptCardProps {
  prompt: Prompt;
  userTier: Tier;
  isFavorited: boolean;
}

export function PromptCard({ prompt, userTier, isFavorited }: PromptCardProps) {
  const [copied, setCopied] = useState(false);
  const { toast } = useToast();
  const { copyPrompt, isLoading: isCopying } = useCopyPrompt();
  const { toggleFavorite, isLoading: isTogglingFavorite } = useFavorite();

  const canAccess = checkTierAccess(userTier, prompt.tier);
  
  const handleCopy = async () => {
    if (!canAccess) {
      toast({
        title: 'Upgrade Required',
        description: `This prompt requires ${prompt.tier} tier or higher.`,
        variant: 'destructive',
      });
      return;
    }
    
    const result = await copyPrompt(prompt.id);
    
    if (result.success) {
      navigator.clipboard.writeText(prompt.promptText);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000);
      
      toast({
        title: 'Copied!',
        description: 'Prompt copied to clipboard.',
      });
    } else if (result.limitReached) {
      toast({
        title: 'Copy Limit Reached',
        description: 'Upgrade to copy more prompts this month.',
        variant: 'destructive',
      });
    }
  };

  const frameworkIcons = {
    react: { bg: '#61DAFB', label: 'React' },
    flutter: { bg: '#02569B', label: 'Flutter' },
    html: { bg: '#E34F26', label: 'HTML' },
    vue: { bg: '#4FC08D', label: 'Vue' },
  };

  return (
    <div className={cn(
      "group bg-card rounded-2xl overflow-hidden border border-border",
      "transition-all duration-200 hover:border-primary hover:shadow-lg hover:shadow-primary/10"
    )}>
      {/* Preview Image */}
      <div className="relative h-36 bg-gradient-to-br from-primary/20 to-secondary/20">
        {prompt.previewImageUrl && (
          <Image
            src={prompt.previewImageUrl}
            alt={prompt.title}
            fill
            className="object-cover"
          />
        )}
        
        {/* Tier Badge */}
        <Badge 
          className="absolute top-3 right-3"
          variant={prompt.tier === 'free' ? 'secondary' : 'default'}
        >
          {prompt.tier.toUpperCase()}
        </Badge>
        
        {/* Generate Button (on hover) */}
        {canAccess && (
          <Button
            size="sm"
            className="absolute bottom-3 right-3 opacity-0 group-hover:opacity-100 transition-opacity"
            onClick={() => router.push(`/generate?prompt=${prompt.id}`)}
          >
            <Sparkles className="w-4 h-4 mr-1" />
            Generate
          </Button>
        )}
      </div>
      
      {/* Content */}
      <div className="p-4">
        <div className="flex justify-between items-start mb-3">
          <div>
            <h4 className="font-semibold text-foreground">{prompt.title}</h4>
            <p className="text-sm text-muted-foreground">{prompt.category.name}</p>
          </div>
          
          {/* Framework Icons */}
          <div className="flex gap-1.5">
            {prompt.frameworks.map((fw) => (
              <div
                key={fw}
                className="w-6 h-6 rounded"
                style={{ backgroundColor: frameworkIcons[fw]?.bg }}
                title={frameworkIcons[fw]?.label}
              />
            ))}
          </div>
        </div>
        
        {/* Actions */}
        <div className="flex gap-2">
          <Button
            className="flex-1"
            onClick={handleCopy}
            disabled={isCopying}
          >
            {copied ? (
              <><Check className="w-4 h-4 mr-1" /> Copied</>
            ) : (
              <><Copy className="w-4 h-4 mr-1" /> Copy</>
            )}
          </Button>
          
          <Button
            variant="outline"
            size="icon"
            onClick={() => toggleFavorite(prompt.id)}
            disabled={isTogglingFavorite}
          >
            <Heart 
              className={cn(
                "w-4 h-4",
                isFavorited && "fill-red-500 text-red-500"
              )} 
            />
          </Button>
        </div>
      </div>
    </div>
  );
}

// src/stores/generator-store.ts

import { create } from 'zustand';

type GeneratorState = 
  | 'IDLE'
  | 'VALIDATING'
  | 'CHECKING_CREDITS'
  | 'GENERATING'
  | 'SUCCESS'
  | 'ERROR';

interface GeneratorStore {
  // State
  state: GeneratorState;
  error: string | null;
  
  // Configuration
  template: string;
  framework: Framework;
  style: Style;
  primaryColor: string;
  darkMode: boolean;
  animationLevel: AnimationLevel;
  customInstructions: string;
  
  // Output
  generatedCode: string | null;
  previewHtml: string | null;
  tokensUsed: number;
  latencyMs: number;
  
  // Actions
  setTemplate: (template: string) => void;
  setFramework: (framework: Framework) => void;
  setStyle: (style: Style) => void;
  setPrimaryColor: (color: string) => void;
  setDarkMode: (enabled: boolean) => void;
  setAnimationLevel: (level: AnimationLevel) => void;
  setCustomInstructions: (instructions: string) => void;
  
  generate: () => Promise<void>;
  regenerate: () => Promise<void>;
  reset: () => void;
}

export const useGeneratorStore = create<GeneratorStore>((set, get) => ({
  // Initial state
  state: 'IDLE',
  error: null,
  template: 'login',
  framework: 'react',
  style: 'glassmorphism',
  primaryColor: '#7C3AED',
  darkMode: true,
  animationLevel: 'subtle',
  customInstructions: '',
  generatedCode: null,
  previewHtml: null,
  tokensUsed: 0,
  latencyMs: 0,
  
  // Setters
  setTemplate: (template) => set({ template }),
  setFramework: (framework) => set({ framework }),
  setStyle: (style) => set({ style }),
  setPrimaryColor: (primaryColor) => set({ primaryColor }),
  setDarkMode: (darkMode) => set({ darkMode }),
  setAnimationLevel: (animationLevel) => set({ animationLevel }),
  setCustomInstructions: (customInstructions) => set({ customInstructions }),
  
  // Generate action
  generate: async () => {
    const state = get();
    
    set({ state: 'VALIDATING', error: null });
    
    // Validate inputs
    if (!state.template) {
      set({ state: 'ERROR', error: 'Please select a template' });
      return;
    }
    
    set({ state: 'CHECKING_CREDITS' });
    
    // Check credits via API
    const creditsRes = await fetch('/api/user/credits');
    const { credits } = await creditsRes.json();
    
    if (credits < 1) {
      set({ state: 'ERROR', error: 'Insufficient credits' });
      return;
    }
    
    set({ state: 'GENERATING' });
    
    try {
      const response = await fetch('/api/generate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          template: state.template,
          framework: state.framework,
          options: {
            style: state.style,
            primaryColor: state.primaryColor,
            darkMode: state.darkMode,
            animationLevel: state.animationLevel,
          },
          customInstructions: state.customInstructions,
        }),
      });
      
      const data = await response.json();
      
      if (!response.ok) {
        throw new Error(data.error || 'Generation failed');
      }
      
      set({
        state: 'SUCCESS',
        generatedCode: data.code,
        previewHtml: data.preview,
        tokensUsed: data.tokensUsed,
        latencyMs: data.latencyMs,
      });
    } catch (error) {
      set({ state: 'ERROR', error: error.message });
    }
  },
  
  regenerate: async () => {
    const state = get();
    set({ state: 'IDLE' });
    await state.generate();
  },
  
  reset: () => set({
    state: 'IDLE',
    error: null,
    generatedCode: null,
    previewHtml: null,
  }),

5.3 Video Player Component

// src/components/learn/video-player.tsx

'use client';

import { useRef, useState, useEffect } from 'react';
import MuxPlayer from '@mux/mux-player-react';
import { useProgress } from '@/hooks/use-progress';

interface VideoPlayerProps {
  playbackId: string;
  lessonId: string;
  initialPosition?: number;
  onComplete: () => void;
}

export function VideoPlayer({
  playbackId,
  lessonId,
  initialPosition = 0,
  onComplete,
}: VideoPlayerProps) {
  const playerRef = useRef<HTMLVideoElement>(null);
  const { saveProgress } = useProgress();
  const [playbackRate, setPlaybackRate] = useState(1);
  
  // Save progress every 10 seconds
  useEffect(() => {
    const interval = setInterval(() => {
      if (playerRef.current) {
        saveProgress({
          lessonId,
          position: Math.floor(playerRef.current.currentTime),
          duration: Math.floor(playerRef.current.duration),
        });
      }
    }, 10000);
    
    return () => clearInterval(interval);
  }, [lessonId, saveProgress]);
  
  // Handle video end
  const handleEnded = () => {
    saveProgress({
      lessonId,
      position: playerRef.current?.duration || 0,
      duration: playerRef.current?.duration || 0,
      completed: true,
    });
    onComplete();
  };

  return (
    <div className="relative aspect-video bg-black rounded-2xl overflow-hidden">
      <MuxPlayer
        ref={playerRef}
        playbackId={playbackId}
        startTime={initialPosition}
        playbackRate={playbackRate}
        onEnded={handleEnded}
        accentColor="#7C3AED"
        metadata={{
          video_title: lessonId,
          viewer_user_id: userId,
        }}
      />
      
      {/* Speed Control */}
      <div className="absolute top-4 right-4 flex gap-2">
        {[0.5, 1, 1.5, 2].map((rate) => (
          <button
            key={rate}
            onClick={() => setPlaybackRate(rate)}
            className={cn(
              "px-3 py-1.5 rounded-lg text-sm font-medium",
              playbackRate === rate
                ? "bg-primary text-white"
                : "bg-black/50 text-white/70 hover:bg-black/70"
            )}
          >
            {rate}x
          </button>
        ))}
      </div>
    </div>
  );
}
